    import os

def add_numbering_to_markdown(markdown_text):
    numbered_lines = []
    title_counts = [0, 0, 0, 0, 0]  # 用于记录每个级别标题的数量
    in_code_block = False

    for line in markdown_text.split('\n'):
        if re.match(r'^\s*```', line):  # 匹配以 ``` 开头的行（代码块开始行）
            if in_code_block:
                numbered_lines.append(line)
                in_code_block = False
            else:
                numbered_lines.append(line)
                in_code_block = True
        elif in_code_block:  # 处理代码块中的行
            numbered_lines.append(line)
        elif re.match(r'^\s*#####\s', line):  # 匹配以 ##### 开头的行（五级标题行）
            title_counts[4] += 1
            line = re.sub(r'^\s*#####\s', f'##### {title_counts[0]}.{title_counts[1]}.{title_counts[2]}.{title_counts[3]}.{title_counts[4]}. ', line)
            numbered_lines.append(line)
        elif re.match(r'^\s*####\s', line):  # 匹配以 #### 开头的行（四级标题行）
            title_counts[3] += 1
            title_counts[4] = 0
            line = re.sub(r'^\s*####\s', f'#### {title_counts[0]}.{title_counts[1]}.{title_counts[2]}.{title_counts[3]}. ', line)
            numbered_lines.append(line)
        elif re.match(r'^\s*###\s', line):  # 匹配以 ### 开头的行（三级标题行univs9.github.io）
            title_counts[2] += 1
            title_counts[3] = 0
            title_counts[4] = 0
            line = re.sub(r'^\s*###\s', f'### {title_counts[0]}.{title_counts[1]}.{title_counts[2]}. ', line)
            numbered_lines.append(line)
        elif re.match(r'^\s*##\s', line):  # 匹配以 ## 开头的行（二级标题行）
            title_counts[1] += 1
            title_counts[2] = 0
            title_counts[3] = 0
            title_counts[4] = 0
            line = re.sub(r'^\s*##\s', f'## {title_counts[0]}.{title_counts[1]}. ', line)
            numbered_lines.append(line)
        elif re.match(r'^\s*#\s', line):  # 匹配以 # 开头的行（一级标题行）
            title_counts[0] += 1
            title_counts[1] = 0
            title_counts[2] = 0
            title_counts[3] = 0
            title_counts[4] = 0
            line = re.sub(r'^\s*#\s', f'# {title_counts[0]}. ', line)
            numbered_lines.append(line)
        else:
            numbered_lines.append(line)

    numbered_text = '\n'.join(numbered_lines)
    return numbered_text
